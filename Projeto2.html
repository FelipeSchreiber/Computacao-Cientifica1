def retangulo(f, a, b, n=1000):
    l, h = np.linspace(a, b, n, endpoint=False, retstep=True)
    soma = np.sum(f(l))
    return soma * h

def trapezio(f, a, b, n=1000):
    l, h = np.linspace(a, b, n, endpoint=False, retstep=True)
    soma = np.sum(f(l))
    correcao = - (f(b)-f(a))/2
    return (soma - correcao) * h

def midpoint(f, a, b, n=1000):
    l, h = np.linspace(a, b, n, endpoint=False, retstep=True)
    mids = l + h/2
    soma = np.sum(f(mids))
    return soma * h

def simpson(f, a, b, n=1000):
    l, h = np.linspace(a, b, n, endpoint=False, retstep=True)
    mids = l + h/2
    soma = 4*np.sum(f(mids)) + 2*np.sum(f(l))
    correcao = - (f(b)-f(a))
    return (soma - correcao) * h/6

A escolha de 0.2*x foi para analisarmos uma funo com uma menor variao quando comparada a exp(x). Podemos ver que o resultados so diferentes, uma vez que para os 4 mtodos temos erros muito parecidos, no havendo tanta diferena na convergncia. Vamos analisar os erros de integrao para cada um dos mtodos:

$ \begin{equation} erro_{retngulo} \leq (b-a)\max_{x \in [a,b]}|f(x)|\\ erro_{trapzio} \leq \frac{5(b-a)h^2\max_{x \in [a,b]}|f^{(2)}(x)|}{12}\\ erro_{ponto mdio} \leq \frac{7(b-a)h^2\max_{x \in [a,b]}|f^{(2)}(x)|}{24}\\ erro_{simpson} \leq \frac{49(b-a)h^4\max_{x \in [a,b]}|f^{(4)}(x)|}{2880} \end{equation} $

Vemos das equaes acima que, caso a funo que queremos integrar no possua derivada quarta, isto $f^{(4)}(x) = 0, \forall x$, ento todas tero um erro associado, exceto para o calculado por simpson.

def simpson2(f, a, b, tol=1e-12):
    n = 1
    simpson1, simpson2 = simpson(f, a, b, n), simpson(f, a, b, 2*n)
    while True:
        l, h = np.linspace(a, b, n, endpoint=False, retstep=True)
        if n != 1:
            dif = simpson1 - simpson2
            if abs(dif) <= tol :
                break
        n=2*n
        simpson1, simpson2 = simpson2, simpson(f, a, b, 2*n)
    return simpson1, n

Questão 3: Dobrando Simpson aos poucos¶

Se usarmos um intervalo de integrao grande, pode ser que a variao da funo em uma parte seja bem menor do que em outra, e da seria mais importante "refinar" o mtodo apenas na parte onde a funo varia mais.

Para fazer isso, vamos usar uma abordagem recursiva:

  1. calculamos a integral no intervalo $[a,b]$ com 1 e com 2 subdivises
  2. se ambas as integrais estiverem suficientemente prximas, terminamos:
    • retornamos a aproximao, e "2" para o nmero de subdivises
  3. seno, calculamos recursivamente a integral em $[a,m]$ e $[m,b]$,
    • e retornamos a soma das integrais, alm da soma do nmero de subdivises para cada metade.

Explique porque, ao calcular recursivamente, importante dividir a tolerncia por 2.

Student's answer Score: 1.0 / 2.0 (Top)

Suponha que comeamos com um erro $\epsilon_0$. Fazemos a avaliao e verificamos que a diferena entre a integral no intervalo [a,b] e a soma das integrais [a,m] + [m,b] superior a tolerncia. Portanto devemos efetuar a chamada recursiva. Caso no dividssimos por 2 a tolerncia, seria possvel que somando os erros da integral da esquerda com a da direita tivssemos um $\epsilon_{final} > \epsilon_0$.

def simpson_rec(f, a, b, tol=1e-8):
    sub_1 = simpson(f,a,b,1)
    m = (a+b)/2
    sub_2 = simpson(f,a,m,1) + simpson(f,m,b,1)
    if(np.abs(sub_1 - sub_2) <= tol):
        return sub_2,2
    approx_left,n = simpson_rec(f, a, m, tol=tol/2) 
    approx_right,n_2 = simpson_rec(f,m,b,tol=tol/2)
    return approx_left + approx_right, n+ n_2

def get_integracao_tol(g, h, a, b,tols, tabela = False):
    err_g = []

def multiint(f, intervals, tol=1e-8):
    tol = tol/len(intervals)
    n=0
    approx = 0
    for interval in intervals:
        r_parc,n_parc = simpson_rec(f,interval[0],interval[1],tol)
        approx += r_parc
        n += n_parc
    return approx, n
